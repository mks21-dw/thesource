(define spiky
  (lambda (n)
    (truncateUntil n (spikyFull n))))

(define truncateUntil
  (lambda (n g)
    (if (>= n (foldl (lambda (new old) (+ old 1)) 0 g))
        g
        (truncateUntil n (truncate g)))))

(define spikyFull
  (lambda (n)
    (if (= n 0)
        '()
        (join (spikyFull (- n 1)) (spike n)))))

(define spike
  (lambda (n)
    (build-list (* n 2) (lambda (x) (+ n 1 (- (abs (- n x))))))))

(define join
  (lambda (g0 g1)
    (if (null? g0)
        g1
        (join (truncate g0)
              (cons (last g0) g1)))))

(define last
  (lambda (g)
    (if (null? (rest g))
        (first g)
        (last (rest g)))))

  (lambda (g)
    (if (null? (rest g))
        '()
        (cons (first g) (truncate (rest g))))))

                                 (* n (factorial (- n 1))

                                / (factorial n) (* (factorial r) (factorial (- n r))))
                 )

                                if (= col 0)
                                   '(1)
                                   (cons (choose row col) (pascalR row (- col 1)))

(define pascalRow (lambda (row)( 
                                pascalR row row
